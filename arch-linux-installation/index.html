Instalasi Arch Linux

  1. Persiapkan bahan di perangkat lain sebagai server penyedia file
    1. archlinux-YYYY.MM.DD-x86_64.iso
    2. fonts.conf
    3. id_ed25519
    4. id_ed25519
  2. Koneksi ke Wi-Fi 
    iwctl station wlan0 connect 'Galaxy A05s'
  3. Partisi, format, dan mounting 
    fdisk /dev/nvme0n1
# Part         ID      Name              Mount point
# nvme0n1p1    uefi    EFI System        /boot
# nvme0n1p2    swap    Linux swap        [SWAP]
# nvme0n1p3    linux   Linux filesystem  /

mkfs.fat -F 32 /dev/nvme0n1p1
mkswap /dev/nvme0n1p2
mkfs.ext4 /dev/nvme0n1p3

mount /dev/nvme0n1p3 /mnt
mkdir /mnt/boot
mount /dev/nvme0n1p1 /mnt/boot
swapon /dev/nvme0n1p2
  4. Mirrorlist 
    curl 'https://archlinux.org/mirrorlist/?country=ID&protocol=http&protocol=https&ip_version=4' > /etc/pacman.d/mirrorlist
  5. Pacstrap 
    pacman -Sy
pacstrap -K /mnt base linux linux-firmware
  6. Generate fstab 
    genfstab -U /mnt >> /mnt/etc/fstab
  7. Chroot 
    arch-chroot /mnt
  8. Atur timezone 
    ln -sf /usr/share/zoneinfo/Asia/Jakarta /etc/localtime
hwclock --systohc
  9. Konfigurasi locale 
    pacman -Syu
pacman -S nano

nano /etc/locale.gen
# Hapus komentar "en_US.UTF-8"
locale-gen

echo LANG=en_US.UTF-8 > /etc/locale.conf
  10. Host 
    echo arch > /etc/hostname

nano /etc/hosts
# Tambahkan baris:
# 127.0.1.1 arch
  11. Membuat user 
    pacman -S sudo
useradd -m -G wheel user
passwd user

EDITOR=nano visudo
# Hapus komentar %wheel ...
  12. Install bootloader 
    bootctl install
  13. Download dan ekstrak Archiso untuk ditambahkan sebagai boot entry 
    # Ganti user ke user
su user
cd ~
mkdir Downloads
cd Downloads
curl -o archlinux-YYYY.MM.DD-x86_64.iso 192.168.xxx.xxx:3000

# Ganti user ke root
exit

mkdir /boot/EFI/archiso
bsdtar -v -x --no-same-permissions --strip-components 1 -f archlinux-YYYY.MM.DD-x86_64.iso -C /boot/EFI/archiso arch

cd /
  14. Konfigurasi entry bootloader
    1. /boot/loader/loader.conf 
      default arch.conf
timeout 4
console-mode max
editor no
    2. /boot/loader/entries/arch.conf 
      title Arch Linux
linux /vmlinuz-linux
initrd /initramfs-linux.img
options root=UUID=xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx rw quiet loglevel=3 systemd.show_status=auto rd.udev.log_level=3
    3. /boot/loader/entries/arch-fallback.conf 
      title Arch Linux (fallback initramfs)
linux /vmlinuz-linux
initrd /initramfs-linux-fallback.img
options root=UUID=xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx rw
    4. /boot/loader/entries/arch-rescue.conf 
      title Arch Linux (rescue system)
linux /EFI/archiso/boot/x86_64/vmlinuz-linux
initrd /EFI/archiso/boot/x86_64/initramfs-linux.img
options archisobasedir=/EFI/archiso archisosearchfilename=/EFI/archiso/boot/x86_64/vmlinuz-linux
  15. Update bootctl 
    bootctl update
  16. Silent boot 
    # /etc/mkinitcpio.conf
# Tambahkan komentar ke: HOOKS=...
# Tambahkan baris:
HOOKS=(base systemd autodetect microcode modconf kms keyboard sd-vconsole block filesystems)

# Regenerate mkinitcpio
mkinitcpio -P
  17. Network Manager dan Gnome 
    pacman -S networkmanager gnome gnome-extra
systemctl enable NetworkManager
systemctl enable gdm
  18. Install software 
    pacman -S firefox zip unzip wget tree man-db audio-sharing libreoffice-still flatpak jdk-openjdk nodejs-lts-jod npm php apache php-apache mariadb git base-devel ttf-roboto ttf-roboto-mono ttf-roboto-mono-nerd noto-fonts noto-fonts-cjk noto-fonts-emoji noto-fonts-extra
  19. Install yay dan paket-paket dari AUR 
    cd /tmp

git clone https://aur.archlinux.org/yay.git
cd yay
makepkg -sirc

yay -S visual-studio-code-bin
  20. Reboot
  21. Atur layout aplikasi
  22. Konfigurasi Visual Studio Code
    1. Mark as done
    2. Hilangkan centang "Show welcome page on startup"
    3. Aktifkan word wrap
  23. Konfigurasi Firefox
    1. Masuk ke Firefox, skip hingga muncul beranda
    2. Ke pengaturan, atur sebagai default
    3. Keluar Firefox dan buka lagi, lakukan hingga widget cuaca muncul, konfigurasi widget, sembunyikan cuaca dan menu
    4. Keluar lalu masuk lagi
    5. Login Google, GitHub, dan WhatsApp
  24. Konfigurasi font
    1. Download fonts.conf dan pindahkan ke ~/.config/fontconfig/fonts.conf
    2. Update font cache 
      fc-cache
    3. Aktifkan font di gnome-tweaks
  25. Reboot
  26. Konfigurasi Git 
    git config --global user.name "[name]"
git config --global user.email "[email]"
git config --global init.defaultBranch main
  27. Konfigurasi GitHub SSH
    1. Download SSH key dari backup 
      curl -o id_ed25519 192.168.xxx.xxx:3000
curl -o id_ed25519.pub 192.168.xxx.xxx:3000
mkdir .ssh
mv id_ed25519 id_ed25519.pub .ssh
    2. Jalankan SSH 
      eval "$(ssh-agent -s)"
    3. Tambahkan SSH key 
      ssh-add ~/.ssh/id_ed25519
    4. Tes koneksi 
      ssh -T git@github.com
  28. Konfigurasi MariaDB
    1. Install database 
      sudo mariadb-install-db --user=mysql --basedir=/usr --datadir=/var/lib/mysql
    2. Jalankan MariaDB dan tambahkan user 
      sudo systemctl start mariadb
sudo mariadb -u root
      CREATE USER 'admin'@'localhost' IDENTIFIED BY '';
GRANT ALL PRIVILEGES ON *.* TO 'admin'@'localhost' WITH GRANT OPTION;
FLUSH PRIVILEGES;
quit;
      sudo systemctl stop mariadb
  29. Konfigurasi PHP 
    # /etc/php/php.ini
# Hilangkan komentar pada baris
extension=mysqli
  30. Konfigurasi Apache 
    # /etc/httpd/conf/httpd.conf
...
#LoadModule mpm_event_module modules/mod_mpm_event.so
LoadModule mpm_prefork_module modules/mod_mpm_prefork.so
...
LoadModule php_module modules/libphp.so
AddHandler php-script .php
...
Include conf/extra/php_module.conf
  31. Finishing (update packages) 
    yay -Syu
  32. Shutdown
